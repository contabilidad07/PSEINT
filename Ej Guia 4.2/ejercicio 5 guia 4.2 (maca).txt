Algoritmo cincoMatriz
	//5.Rellenar una matriz, de 3 x 3, con una palabra de 9 de longitud, pedida por el usuario,
	//encontrando la manera de que la frase se muestre de manera continua en la matriz. 
	//Por ejemplo, si tenemos la palabra habilidad, nuestra matriz se debería ver así:
	//   [H]   [A]   [B]
	//  [I]   [L]   [I]
	// [D]   [A]   [D]
	//Nota: recordar el uso de la función Subcadena()
	
	Definir palabra, matriz Como Caracter
	Dimension matriz[3,3]

	hacer
	Mostrar "Ingrese una palabra de 9 digitos"
	leer palabra
	palabra = Mayusculas(palabra)
	Mientras Que longitud(palabra) <> 9

Seleccion[matriz,palabra]
FinAlgoritmo

SubProceso Seleccion[matriz,palabra]
	Definir i,j,contador Como Entero
	contador=0 //es para ir tomando las posiciones en la palabra, desde la posicion 0 hasta la 9
	para i=0 Hasta 2 Hacer
		para j=0 Hasta 2 Hacer
			matriz(i,j)= Subcadena(palabra,contador,contador)
			contador= contador +1 //para que vaya sumando las posiciones hasta que llegue todo al 9
			escribir " [ " matriz[i,j] " ] " Sin Saltar
			
		FinPara
		Escribir "" 
	FinPara
FinSubProceso
